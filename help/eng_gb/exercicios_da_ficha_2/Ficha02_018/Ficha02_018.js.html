Operators with multi instructions

Build an algorithm that prompts the user for a real number and an integer and then prints the result of the five operators: + Sum - Subtraction * Multiplication / Division ^ Potentiation

JavaScript

function main() {
    document.write("Real number : " );// print text
    n1 = parseFloat(prompt("Real number : " ,"0.0") );
    
    document.write(n1 + "\n");// print input 
    document.write("Integer  : " );// print text
    n2 = parseInt(prompt("Integer  : " ,"0") );
    
    document.write(n2 + "\n");// print input 
    var r1 = 0 ;
    var r2 = 0 ;
    var r3 = 0 ;
    var r4 = 0 ;
    r1 = n1 + n2 ;
    r2 = n1 - n2 ;
    r3 = n1 * n2 ;
    r4 = n1 / n2 ;
    document.write(n1 + " + " + n2 + " = " + r1 );
    document.write("\n" + n1 + " - " + n2 + " = " + r2 );
    document.write("\n" + n1 + " * " + n2 + " = " + r3 );
    document.write("\n" + n1 + " / " + n2 + " = " + r4 );
}

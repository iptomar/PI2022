Operators with multi instructions

Build an algorithm that prompts the user for a real number and an integer and then prints the result of the five operators: + Sum - Subtraction * Multiplication / Division ^ Potentiation

PHP 5.3 +

<?php

function main() {
    echo  "Real number : " ;
    $n1 = floatval(isset($_POST["n1"])? $_POST["n1"] : isset($_GET["n1"]) ? $_GET["n1"] : 0);
    
    echo $n1 . "<br/>";// print input 
    echo  "Integer  : " ;
    $n2 = intval(isset($_POST["n2"])? $_POST["n2"] : isset($_GET["n2"]) ? $_GET["n2"] : 0);
    
    echo $n2 . "<br/>";// print input 
    $r1 = 0 ;
    $r2 = 0 ;
    $r3 = 0 ;
    $r4 = 0 ;
    $r1 = $n1 + $n2 ;
    $r2 = $n1 - $n2 ;
    $r3 = $n1 * $n2 ;
    $r4 = $n1 / $n2 ;
    echo $n1  .  " + " . $n2  .  " = " . $r1  ;

}

}

function line(n) {
    for( var i = 0; i < n; i = i + 1 ) {
        document.write("*" );
    }
    document.write("\n" );
}
//::::::::::::::::::::::::::::::::::::::::::::::::::
function triangle(height) {
    for( var i = 1; i < height + 1; i = i + 1 ) {
        line(i ) ;
    }
}

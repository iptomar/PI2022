Minimum Grade

Build an algorithm that asks the user for the staff grade of 5 students. Then print out the minimum score.

C Language c11

#include <stdio.h>
#include <string.h>
#include <math.h>
//Main Function
int main() {
    int minimum = 999 ;
    int i = 0 ;
    for( int i = 1; i < 6; i = i + 1 ) {
        printf("%dº grade : ",i );
        int grade;
        scanf("%d", &grade);
        if(grade < minimum ){
            minimum = grade ;
        }
    }
    printf("Nota mínima = %d",minimum );
    return 0;
}

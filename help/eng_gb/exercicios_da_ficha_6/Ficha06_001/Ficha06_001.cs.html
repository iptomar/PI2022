using System.IO;
using System;
public class Ficha06_001 {

    //Main Function
    static void Main() {
        for( int n = 1; n < 11; n = n + 1 ) {
            Console.Write("\nTabuada do " + n + "\n" );
            for( int i = 1; i < 11; i = i + 1 ) {
                Console.Write(n + " x " + i + " = " + i * n + "\n" );
            }
        }
    }

}

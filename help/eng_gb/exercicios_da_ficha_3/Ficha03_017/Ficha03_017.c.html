Number of Characters

Build an algorithm that prints the number of characters of a name entered by the user.

C Language c11

#include <stdio.h>
#include <string.h>
#include <math.h>
#define length( a ) ( sizeof( a ) / sizeof( a[0] ))
//Main Function
int main() {
    printf("Name : ");
    char name[256];
    fgets(name,256, stdin);
    printf("Number of characters = %d",length( name ) );
    return 0;
}

using System;
public class Ficha07_005 {

    //Main Function
    static void Main() {
        Console.Write("Productory\n" );
        int product = 1 ;
        for( int i = 1; i < 5; i = i + 1 ) {
            Console.Write(product + " x " + i );
            product *=i;
            Console.Write(" = " + product + "\n" );
        }
        Console.Write("\nTotal = " + product );
    }

}

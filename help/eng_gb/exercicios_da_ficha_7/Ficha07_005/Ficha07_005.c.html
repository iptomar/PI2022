//Main Function
int main() {
    printf("Productory\n");
    int product = 1 ;
    int i = 0 ;
    for( int i = 1; i < 5; i = i + 1 ) {
        printf("%d x %d",product ,i );
        product *=i;
        printf(" = %d\n",product );
    }
    printf("\nTotal = %d",product );
    return 0;
}

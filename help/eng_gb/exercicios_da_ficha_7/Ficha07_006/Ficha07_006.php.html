Odd-Productory

Build a program that calculates the value of the formula. The value of n must be entered by the user.

PHP 5.3 +

<?php

//Main Function
function main() {
    echo  "Number of terms : " ;
    $n = intval(isset($_POST["n"])? $_POST["n"] : isset($_GET["n"]) ? $_GET["n"] : 0);
    
    echo $n . "<br/>";// print input 
    $product = 1 ;
    for( $i = 1; $i < $n + 1; $i = $i + 1 ) {
        $f = 2 * $i - 1 ;
        $product = $product * $f ;
    }
    echo  "Result = " . $product  ;
}

main();

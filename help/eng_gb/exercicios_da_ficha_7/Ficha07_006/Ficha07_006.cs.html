using System.IO;
using System;
public class Ficha07_006 {

    //Main Function
    static void Main() {
        Console.Write("Number of terms : " );
        int n = int.Parse(Console.ReadLine());
        int product = 1 ;
        for( int i = 1; i < n + 1; i = i + 1 ) {
            int f = 2 * i - 1 ;
            product *=f;
        }
        Console.Write("Result = " + product );
    }

}

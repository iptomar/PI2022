public class Ficha07_001 {

    //Main Function
    public static void main( String[] args) {
        System.out.print("Summation\n" );
        int sum = 0 ;
        for( int i = 1; i < 6; i = i + 1 ) {
            System.out.print(sum + " + " + i );
            sum++;
            System.out.print(" \t = " + sum + "\n" );
        }
        System.out.print("\nTotal = " + sum );
    }

}

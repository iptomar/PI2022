Passing from a array to a function
Flowchart Pseudocode 
begin MainProgram
    define integer array[8][4]
    for integer i from 0 to elementsOf(array) step 1
        execute fill(array[i])
        for integer j from 0 to length(array[i]) step 1
            write "\t"+array[i][j]
        end for
        write "\n"
    end for
end MainProgram

function void fill(integer v[])
    for integer i from 0 to elementsOf(v) step 1
        execute v[i] = int(rand()*10)
    end for
end fill

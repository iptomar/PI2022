#include <stdio.h>
#include <string.h>
#include <math.h>
//Main Function
int main() {
    printf("Número \t: ");
    int num;
    scanf("%d", &num);
    if(num >= 0 ){
        if(num > 0 ){
            printf("Positivo");
        }else{
            printf("Nulo");
        }
    }else{
        printf("Negativo");
    }
    return 0;
}

Operadores Relacionais

Problema Fluxograma Algorithmi
O resultado de uma expressão que contenha um operador relacional é SEMPRE um valor lógico Operadores relacionais: == igual != diferente > maior >= maior ou igual < menor <= menor ou igual

 
    begin MainProgram
        read integer n1 "Primeiro valor \t: "
        read integer n2 "Segundo valor \t\t: "
        define boolean igual = n1 == n2
        write "\n" + n1 + "\t==\t" + n2 + "\t=\t" + igual
        define boolean diferente = n1 != n2
        write "\n" + n1 + "\t!=\t" + n2 + "\t=\t" + diferente
        define boolean compara = n1 > n2
        write "\n" + n1 + "\t>\t" + n2 + "\t=\t" + compara
        execute compara = n1 >= n2
        write "\n" + n1 + "\t>=\t" + n2 + "\t=\t" + compara
        write "\n" + n1 + "\t<\t" + n2 + "\t=\t" + (n1 < n2)
        write "\n" + n1 + "\t<=\t" + n2 + "\t=\t" + (n1 <= n2)
    end MainProgram

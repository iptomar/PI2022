Construa um algoritmo que verifique se um numero introduzido pelo utilizador e positivo, negativo ou nulo.

JavaScript

//Main Function
function main() {
    document.write("Número \t: " );// print text
    num = parseInt(prompt("Número \t: " ,"0") );
    
    document.write(num + "\n");// print input 
    if(num == 0 ){
        document.write("Nulo" );
    }else{
        if(num > 0 ){
            document.write("Positivo" );
        }else{
            document.write("Negativo" );
        }
    }
}

using System;
public class Ficha04_005 {

    //Main Function
    static void Main() {
        Console.Write("Número \t: " );
        int num = int.Parse(Console.ReadLine());
        if(num == 0 ){
            Console.Write("Nulo" );
        }else{
            if(num > 0 ){
                Console.Write("Positivo" );
            }else{
                Console.Write("Negativo" );
            }
        }
    }

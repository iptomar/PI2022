Divisores de um numero

Construa um algoritmo que imprima na consola os divisores de um numero. O numero deve ser introduzido pelo utilizador.

C#

using System.IO;
using System;
public class Ficha09_016 {

    //Main Function
    static void Main() {
        Console.Write("Numero : " );
        int num = int.Parse(Console.ReadLine());
        int i = 1 ;
        Console.Write("Divisores : " );
        while(i < num ){
            if(num % i == 0 ){
                Console.Write(i + " " );
            }
            i++;
        }
    }

}

sequência de Fibonacci

Construa um algoritmo que imprima na consola os números da sequência de fibonacci que são menores que 100. https://en.wikipedia.org/wiki/Fibonacci_number

Java

// Programmer           Ant0ni0 M@ns0 - Ant0ni0 M@ns0

public class Ficha09_007 {

    //Main Function
    public static void main( String[] args) {
        int f0 = 1 ;
        int f1 = 1 ;
        System.out.print(f0 + "\n" + f1 + "\n" );
        while(f0 + f1 < 100 ){
            int fn = f0 + f1 ;
            System.out.print(fn + "\n" );
            f1 = f0 ;
            f0 = fn ;
        }
    }

}

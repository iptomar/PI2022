Bits de um número

Contrua um algoritmo para imprimir na consola os bits de um numero inteiro.

JavaScript

// Programmer           Ant0ni0 M@ns0 - Ant0ni0 M@ns0

//Main Function
function main() {
    document.write("Numero : " );// print text
    num = parseInt(prompt("Numero : " ,"0") );
    
    document.write(num + "\n");// print input 
    document.write("Bits = " );
    while(num > 0 ){
        document.write(num % 2 + " " );
        num /=2;
    }
}

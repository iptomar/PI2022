Soma de sequência de Fibonaci

Construa um algoritmo que imprima na consola os números da sequência de fibonacci que são menores que 100. De seguida deve imprimir a sua soma. https://en.wikipedia.org/wiki/Fibonacci_number

C#

using System.IO;
using System;
public class Ficha09_008 {

    //Main Function
    static void Main() {
        Console.Write("Soma da sequência de fibonacci\n " );
        int f0 = 1 ;
        int f1 = 1 ;
        Console.Write(f0 + " " + f1 );
        int soma = 2 ;
        while(f0 + f1 < 100 ){
            int fn = f0 + f1 ;
            soma +=fn;
            Console.Write(" " + fn );
            f1 = f0 ;
            f0 = fn ;
        }
        Console.Write("\nSoma = " + soma );
    }

}

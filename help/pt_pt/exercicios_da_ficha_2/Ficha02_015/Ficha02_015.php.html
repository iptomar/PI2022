Operadores inteiros

Construa um algoritmo que solicite ao utilizador dois números inteiros e de seguida imprima o resultado dos seis operadores: + Soma - Subtração * multiplicação / Divisão % Resto da divisão ^ Potenciação

PHP 5.3 +

<?php
// Programador          Ant0ni0 M@ns0 - Ant0ni0 M@ns0

//Main Function
function main() {
    echo  "1º número :" ;
    $n1 = intval(isset($_POST["n1"])? $_POST["n1"] : isset($_GET["n1"]) ? $_GET["n1"] : 0);
    
    echo $n1 . "<br/>";// print input 
    echo  "2º número :" ;
    $n2 = intval(isset($_POST["n2"])? $_POST["n2"] : isset($_GET["n2"]) ? $_GET["n2"] : 0);
    
    echo $n2 . "<br/>";// print input 
    echo $n1  .  " + " . $n2  .  " = " . ( ( $n1 + $n2 )  ) ;
    echo  "\n" . $n1  .  " - " . $n2  .  " = " . ( ( $n1 - $n2 )  ) ;
    echo  "\n" . $n1  .  " * " . $n2  .  " = " . ( ( $n1 * $n2 )  ) ;
    echo  "\n" . $n1  .  " / " . $n2  .  " = " . ( ( $n1 / $n2 )  ) ;
    echo  "\n" . $n1  .  " % " . $n2  .  " = " . ( ( $n1 % $n2 )  ) ;
    echo  "\n" . $n1  .  " ^ " . $n2  .  " = " . ( ( $n1 ^ $n2 )  ) ;
}

main();

Construa um algoritmos que some dois números inteiros introduzidos pelo utilizador
Fluxograma Algorithmi 
    begin MainProgram
        define integer n1 = 0,n2 = 0
        read n1 "1º número : "
        read n2 "2º número : "
        define integer soma = n1 + n2
        write n1 + " + " + n2 + " = " + soma
    end MainProgram

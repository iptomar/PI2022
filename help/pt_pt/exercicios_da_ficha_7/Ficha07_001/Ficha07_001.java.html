public class Ficha07_001 {

    //Main Function
    public static void main( String[] args) {
        System.out.print("Somatório\n" );
        int soma = 0 ;
        for( int i = 1; i < 6; i = i + 1 ) {
            System.out.print(soma + " + " + i );
            soma +=i;
            System.out.print(" \t = " + soma + "\n" );
        }
        System.out.print("\nTotal = " + soma );
    }

}

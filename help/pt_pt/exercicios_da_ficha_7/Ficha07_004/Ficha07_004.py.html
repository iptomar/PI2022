Somatório 5 pares

Soma o primeiros 5 pares NOTA: par(i) = 2*i

Python 3

print( "Somatório Pares [0,10]")
soma = 0 
for i in range(1, 6, 1 ):
    f = 2 * i 
    print(soma ,  " + ", f  , end="")
    soma = soma + f
    print( " = ", soma)
print( "\nSoma = ", soma  , end="")

Somatório Fórmula

Construa um algoritmo que calcule a soma dos primeiro 5 números impares. NOTA: impar(i) = 2*i - 1

C Language c11

// Programmer           Ant0ni0 M@ns0 - Ant0ni0 M@ns0

#include <stdio.h>
#include <string.h>
#include <math.h>
//Main Function
int main() {
    printf("Somatório Fórmula\n");
    int soma = 0 ;
    int i = 0 ;
    for( int i = 1; i < 6; i = i + 1 ) {
        int f = 2 * i - 1 ;
        printf("%d + %d",soma ,f );
        soma +=f;
        printf(" \t= %d\n",soma );
    }
    printf("\nSoma = %d",soma );
    return 0;
}

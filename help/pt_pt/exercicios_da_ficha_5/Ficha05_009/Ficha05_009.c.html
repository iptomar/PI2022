Pares num intervalo - Calcular

Escreva um algoritmo que solicite ao utilizador os limites de um intervalo. De seguida imprima todos os números pares desse intervalo incluindo os limites.

#include <stdio.h>
#include <string.h>
#include <math.h>
//Main Function
int main() {
    printf("Minimo : ");
    int imin;
    scanf("%d", &imin);
    printf("Maximo : ");
    int imax;
    scanf("%d", &imax);
    if(imin % 2 != 0 ){
        imin++;
    }
    int i = 0 ;
    for( int i = imin; i < imax + 1; i = i + 2 ) {
        printf("%d ",i );
    }
    return 0;
}

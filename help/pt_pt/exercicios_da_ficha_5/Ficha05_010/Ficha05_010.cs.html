Nota Mínima

Contrua um algoritmo que solicite ao utilizador a nota de pauta de 5 alunos. De seguida imprima a nota mínima.

C#

// Programmer           Ant0ni0 M@ns0 - Ant0ni0 M@ns0

using System.IO;
using System;
public class Ficha05_010 {

    //Main Function
    static void Main() {
        int minimo = 999 ;
        for( int i = 1; i < 6; i = i + 1 ) {
            Console.Write(i + "º nota : " );
            int nota = int.Parse(Console.ReadLine());
            if(nota < minimo ){
                minimo = nota ;
            }
        }
        Console.Write("Nota mínima = " + minimo );
    }

}

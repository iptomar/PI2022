Altura de uma Pessoa

Construa um algoritmo que solicite ao utilizador a altura de uma pessoa em metros. A altura de uma pessoa varia entre 20cm e 3 metros.

C#

// Programmer           Ant0ni0 M@ns0 - Ant0ni0 M@ns0

using System.IO;
using System;
public class Ficha08_005 {

    //Main Function
    static void Main() {
        double altura = 0.0 ;
        do{
            Console.Write("Altura : " );
            altura = double.Parse(Console.ReadLine());
        }while(altura < 0.2 || altura > 3.0 );
        Console.Write("Altura = " + altura );
    }

}

Construa um algoritmo que solicite ao utilizar um número de de seguida imprima o seu logaritmo natural.

function main() {
    document.write("Valor de X\t: " );// print text
    x = parseFloat(prompt("Valor de X\t: " ,"0.0") );
    
    document.write(x + "\n");// print input 
    var y = Math.log( x ) ;
    document.write("\nln(X) \t\t= " + y );
}

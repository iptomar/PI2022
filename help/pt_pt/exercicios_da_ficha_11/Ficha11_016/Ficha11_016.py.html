Copa

Construa um algoritmo que imprima na consola a figura seguinte. A altura e o caracter devem ser introduzidos pelo utilizador.

#User Defined Functions Code
#::::::::::::::::::::::::::::::::::::::::::::::::::
def linhaCopa(sp, chars, txt):
    for i in range(0, sp, 1 ):
        print( " " , end="")
    for i in range(0, chars, 1 ):
        print(txt  , end="")
    print()

#::::::::::::::::::::::::::::::::::::::::::::::::::
def copa(alt, txt):
    for i in range(1, alt + 1, 1 ):
        sp = alt - i 
        chrs1 = i * 2 - 1 
        linhaCopa(sp , chrs1 , txt )

if __name__ == "__main__":
    alt =  int(input("Altura   : "))
    char =  str(input("Caractér : "))
    copa(alt , char )

Quadrado de Asteriscos

Construa um algoritmo que imprima o quadrado da figura. A altura deve ser introduzida pelo utilizador. Utilize a funcao: vazio linha(inteiro n) Escreva a funcao: vazio Quadrado(inteiro n)

JavaScript

// Programador           -

//Main Function
function main() {
    document.write("Altura : " );// print text
    altura = parseInt(prompt("Altura : " ,"0") );
    
    document.write(altura + "\n");// print input 
    quadrado(altura ) ;
}

//::::::::::::::::::::::::::::::::::::::::::::::::::
//User Defined Functions Code
//::::::::::::::::::::::::::::::::::::::::::::::::::
function linha(n) {
    for( var i = 0; i < n; i = i + 1 ) {
        document.write("*" );
    }
    document.write("\n" );
}
//::::::::::::::::::::::::::::::::::::::::::::::::::
function quadrado(altura) {
    for( var i = 1; i < altura + 1; i = i + 1 ) {
        linha(altura ) ;
    }
}

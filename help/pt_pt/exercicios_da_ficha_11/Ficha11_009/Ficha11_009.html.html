        function main() {

        document.write(altura + "\n");// print input

    //::::::::::::::::::::::::::::::::::::::::::::::::::
    //User Defined Functions Code

    function linha(n) {
        for( var i = 0; i < n; i = i + 1 ) {
            document.write("*" );
        }
        document.write("\n" );
    }
    //::::::::::::::::::::::::::::::::::::::::::::::::::
    function triangulo(altura) {
        for( var i = 1; i < altura + 1; i = i + 1 ) {
            linha(i ) ;
        }
    }

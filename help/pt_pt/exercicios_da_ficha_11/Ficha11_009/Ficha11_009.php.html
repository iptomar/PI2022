Triângulo - linha e triangulo

Construa um programa que imprima o triangulo da figura. A altura deve ser introduzida pelo utilizador. Escreva a funcao: vazio triangulo(inteiro altura) Utilize a funcao: vazio linha(inteiro n)

PHP 5.3 +

<?php
// Programador           -

//Main Function
function main() {
    echo  "Altura : " ;
    $altura = intval(isset($_POST["altura"])? $_POST["altura"] : isset($_GET["altura"]) ? $_GET["altura"] : 0);
    
    echo $altura . "<br/>";// print input 
    triangulo($altura ) ;
}

//::::::::::::::::::::::::::::::::::::::::::::::::::
//User Defined Functions Code
//::::::::::::::::::::::::::::::::::::::::::::::::::
function linha($n) {
    for( $i = 0; $i < $n; $i = $i + 1 ) {
        echo  "*" ;
    }
    echo  "\n" ;
}
//::::::::::::::::::::::::::::::::::::::::::::::::::
function triangulo($altura) {
    for( $i = 1; $i < $altura + 1; $i = $i + 1 ) {
        linha($i ) ;
    }
}

main();

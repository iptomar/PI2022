Triângulo - linha e triangulo

Construa um programa que imprima o triangulo da figura. A altura deve ser introduzida pelo utilizador. Escreva a funcao: vazio triangulo(inteiro altura) Utilize a funcao: vazio linha(inteiro n)

Java

// Programador           -

import java.util.Scanner;

public class Ficha11_009 {

    static Scanner keyb = new Scanner(System.in);//keyboard

    //Main Function
    public static void main( String[] args) {
        System.out.print("Altura : " );
        int altura = Integer.valueOf(keyb.nextLine());
        triangulo(altura ) ;
    }

    //::::::::::::::::::::::::::::::::::::::::::::::::::
    //User Defined Functions Code
    //::::::::::::::::::::::::::::::::::::::::::::::::::
    public static void linha( int n){
        for( int i = 0; i < n; i = i + 1 ) {
            System.out.print("*" );
        }
        System.out.print("\n" );
    }
    //::::::::::::::::::::::::::::::::::::::::::::::::::
    public static void triangulo( int altura){
        for( int i = 1; i < altura + 1; i = i + 1 ) {
            linha(i ) ;
        }
    }

}

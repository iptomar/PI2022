void linhaC( int n, char * c ){
    int i = 0 ;
    for( int i = 0; i < n; i = i + 1 ) {
        printf("%s ",c );
    }
    printf("\n");
}
//::::::::::::::::::::::::::::::::::::::::::::::::::
void trianguloC( int altura, char * c ){
    int i = 0 ;
    for( int i = 0; i < altura; i = i + 1 ) {
        linhaC(altura - i , c ) ;
    }
}

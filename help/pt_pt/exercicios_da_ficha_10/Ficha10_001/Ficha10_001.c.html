Ciclo infinito - Ler número positivo

Escreva um programa que solicite ao utilizador um número positivo. Se o utilizador introduzir um valor errado deve ser impressa uma mensagem de erro.

C Language c11

// Programador          Ant0ni0 M@ns0 - Ant0ni0 M@ns0

#include <stdio.h>
#include <string.h>
#include <math.h>
//Main Function
int main() {
    int i = 0 ;
    while(1 ){
        printf("Número positvo : ");
        scanf("%d", &i);
        if(i > 0 ){
            break;
        }
        printf("Erro na leitura\n\n");
    }
    printf("Valor introduzido = %d",i );
    return 0;
}
